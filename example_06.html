Bootstrap 3 Form from Zend Framework 1 Example 6. Disabled state

<?php
class Application_Form_Example6 extends Twitter_Bootstrap3_Form_Vertical
{
    public function init()
    {
        $this->addElement('text', 'disabledTextInput', array(
            'label' => 'Disabled input',
            'placeholder' => 'Disabled input',
        ));
        $this->addElement('select', 'disabledSelect', array(
            'label' => 'Disabled select menu',
            'multiOptions' => array(
                '1' => 'Disabled select',
            ),
        ));
        $this->addElement('checkbox', 'disabledCheckbox', array(
            'label' => 'Can\'t check this',
        ));
        $this->addElement('submit', 'disabledSubmit', array(
            'label' => 'Submit',
        ));
        $this->addDisplayGroup(array(
            'disabledTextInput', 'disabledSelect', 'disabledCheckbox', 'disabledSubmit'
        ), 'disabled', array(
            'disabled' => 'disabled',
        ));
    }
}
