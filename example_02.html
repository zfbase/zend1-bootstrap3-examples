Bootstrap 3 Form from Zend Framework 1 Example 2. Inline form

<?php
class Application_Form_Example2 extends Twitter_Bootstrap3_Form_Inline
{
    public function init()
    {
        $this->addElement('email', 'email', array(
            'label' => 'Email address',
            'placeholder' => 'Enter email',
        ));
        $this->addElement('password', 'password', array(
            'label' => 'Password',
            'placeholder' => 'Password',
        ));
        $this->addElement('checkbox', 'checkbox', array(
            'label' => 'Remember me',
        ));
        $this->addElement('submit', 'submit', array(
            'label' => 'Sign in',
            'class' => 'btn-default',
        ));
    }
}
