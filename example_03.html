<?php
class Application_Form_Example3 extends Twitter_Bootstrap3_Form_Inline
{
    public function init()
    {
        $this->init_v1();
    }
    public function init_v1()
    {
        $this->addElement('email', 'email', array(
            'label' => 'Amount (in dollars)',
            'placeholder' => 'Amount',
            'addon_append' => '.00',
            'addon_prepend' => '$',
        ));
        $this->addElement('submit', 'submit', array(
            'label' => 'Transfer cash',
        ));
    }
    public function init_v2()
    {
        $this->addElement('email', 'email', array(
            'label' => 'Amount (in dollars)',
            'placeholder' => 'Amount',
        ));
        $this->getElement('email')->getDecorator('Addon')
            ->setOption('prepend', '$')
            ->setOption('append', '.00');
        $this->addElement('submit', 'submit', array(
            'label' => 'Transfer cash',
        ));
    }
}
